    \newcommand\n{5}
    \begin{tikzpicture}
        \node[vertex] (N-0) at (0:0) {};
        \foreach \number in {1,...,\n}{
            \node[vertex] (N-\number) at ({\number*(360/\n)}:5.4cm) {};
        }

        \draw (N-0) -- (N-1);
        \draw (N-0) -- (N-2);
        \draw (N-0) -- (N-3);
        \draw (N-0) -- (N-4);
